#include<iostream>
using namespace  std;
int reverse(int num){
    int rev = 0;
    while(num){
    rev = rev*10 + (num%10);
    num/=10;
    }
    return rev;
}

int main(){
    int number;
    cout<<"Enter a Number : ";
    cin>>number;
    int num_rev =  reverse(number);
    while(number){
        if((num_rev%10) == (number)%10){
            num_rev/=10;
            number/=10;
        }
        else{
            cout<<"Not a Palindrom Number";
            return 1;
        }
    }
    if(!number){
        cout<<"Palindrom Number";
    }
    return 0;
}
