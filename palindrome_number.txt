#include<iostream>
using namespace  std;
int reverse(int num){
    int rev = 0;
    while(num){
    rev = rev*10 + (num%10);
    num/=10;
    }
    return rev;
}

int main(){
    int number;
    cout<<"Enter a Number : ";
    cin>>number;
    int num_rev =  reverse(number);
    if(num_rev == number){
        cout<<"Palindrom number";
    }
    else{
        cout<<"Not a Palindrom number";
    }
    return 0;
}
